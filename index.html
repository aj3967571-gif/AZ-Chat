async function sendMessage() { const text = userInput.value.trim(); if(!text) return; // User message display messagesDiv.innerHTML += `
${text}
`; userInput.value = ""; // Show Typing... const typingId = "msg-" + Date.now(); messagesDiv.innerHTML += `
Typing...
`; messagesDiv.scrollTop = messagesDiv.scrollHeight; try { // Aapka Vercel link jo screenshot mein dikh raha hai const response = await fetch("https://az-chat-smoky.vercel.app/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: text }) }); const data = await response.json(); document.getElementById(typingId).innerText = data.reply; } catch (error) { document.getElementById(typingId).innerText = "Server error! Please check back later."; console.error("Error:", error); } }
